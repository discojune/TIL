def solution(s):
    from collections import deque

    s_deque = deque(s)
    
    closed_p = 0
    while s_deque:
        p = s_deque.popleft()
        
        if closed_p >= 0 and p == "(":
            closed_p += 1
        elif p == ")":
            closed_p -= 1
            
    if not closed_p:
        return True
    else:
        return False
