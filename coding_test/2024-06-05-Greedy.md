```python
def solution(people, limit):
    from collections import deque
    
    answer = 0
    
    people.sort()
    
    person_1_idx = 0
    person_2_idx = len(people) - 1
    
    while person_2_idx >= person_1_idx:
        person_1 = people[person_1_idx]
        person_2 = people[person_2_idx]
        
        if (person_1 + person_2) <= limit:
            person_1_idx += 1
            person_2_idx -= 1
            
            answer += 1
            
        else:
            person_2_idx -= 1
            answer += 1
        

    return answer

```
